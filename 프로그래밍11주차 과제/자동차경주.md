자동차경주
```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <conio.h> //콘솔창에서 제어를 할수있게 해주는 함수를 위해 선언

void disp_car();

int main(void){
    int car1_dist=0, car2_dist=0; //각각 난수를 저장할 자동차 선언

    srand((unsigned)time(NULL));

    for(int i = 1; i <= 5; i++){ //총 5번 이동(주행시간)
        car1_dist += rand()%100;
        car2_dist += rand()%100; //car에 난수값을 저장
        disp_car(1, car1_dist);
        disp_car(2, car2_dist); //차의 번호와 차의 저장된 난수값을 넣는다.
        printf("------------------------|GOAL\n"); 
        _getch(); //반복문을 제어하여 콘솔창에서 입력을 받아야 다음으로 넘어간다.
    }
    return 0;
}

void disp_car(int car_number, int distance){ //void 는 리턴값이 필요 없다.
    printf("CAR #%d:", car_number); //받은 car를 구분하기 위해 1, 2를 출력
    for(int i = 1; i <= distance/10; i++){ //자동차의 저장된 난수 만큼 *을 출력
        printf("*");
    }
    printf("\n");
}
```
