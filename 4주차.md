# 4주차 c언어 연습

```c
#include <stdio.h>

int main(void)
{
    int item, money, cheon, obaek, baek, a;

    printf("물건 값을 입력하시오: ");
    scanf("%d", &item);

    printf("투입한 금액을 입력하시오:");
    scanf("%d", &money);
    
    printf("거스름돈은 다음과 같습니다.\n");
    a = money - item; //거스름돈

    cheon = a / 1000; //천원짜리
    a %= 1000;// 나머지 돈
    printf("천원권: %d장\n", cheon);

    obaek = a / 500; //오백원짜리
    a %= 500; //나머지돈
    printf("오백원 동전: %d개\n", obaek);

    baek = a / 100; //백원짜리
    printf("백원 동전: %d개\n", baek);

    return 0;
}
```
